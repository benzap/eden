*eden* is an embedded scripting language in clojure for manipulating
 EDN data.

* Examples

#+BEGIN_SRC clojure

;; infix operators

2 + 2 == 4 ;; true

(2 + 2) * 4 == 9 ;; false

;; variables are global

num = 42

s = "string"

y = {:x 123}

local x = 5 ;; local variables

;; while loops

while num < 50 do
  num = num + 1
end

;; if clauses

if num > 40 then
  println ("over 40")
elseif s ~= 'walternate' then
  print ("not over 40\n") 
else
  print ("Winter is coming")
end


;; comparator logic
aBool = true

if not aBool then print ("twas false") end

ans = aBoolValue and "yes" or "no"

;; for loops

karlSum = 0
for i = 1 , 100 do
  karlSum = karlSum + i
end

;; repeat until

repeat
  num = num - 1
until num == 0

;; functions

function fib (n)
  if n < 2 then return n end
  return fib (n - 2) + fib (n - 1)
end
